Table Users {
  // This is the apps Users document that will hold all of the data and fields related
  // To each user that has signed up to the app
  _id string [pk]
  email string [unique]
  hash_password string
  created_at ISO_Time 
  username string
  profilePic string // Will redirect to pic location
  isAdmin bool // True || False
  isDeleted bool
  mainCurrency Currency // !!! The currency the app will use

  Financial_Account LIST_Embedded_Object
}

// START-------- Financial_Accounts --------START \\
Table Currency {
  id MONGO_ID [pk]
  currencyName string
  currencySymbol char // '$' / '₪'..
  currencyCode string // "USD" / "ILS"..
}

Table Balance {
  id MONGO_ID [pk]
  currency Currency
  amount double
}

// Wallet / Bank Acc..
Table Financial_Account {
  id MONGO_ID [pk]
  name string
  balances LIST_Balance
}

Table Bank_Account { // Type of Financial_Account
  id MONGO_ID [pk]
  acoount_number long 
  branch_location string
  branch_number long
  credit_cards LIST_Credit_Cards [note: "Cards that belong to Acc"]
}
Table Credit_Card {
  id MONGO_ID [pk]
  issuer string
  name string 
  limit double
  dueDate datetime
  transactionRecord LIST_transactionRecord
}

// END---------- Financial_Accounts ----------END \\


// START-------- Transactions --------START \\

Table transactionRecord { // CURRENT
  id MONGO_ID [pk]
  date datetime
  from string
  to string
  transactionsList LIST_transaction // The transactionsList of how the main transaction is split up
  payments int [note: "תשלומים"]
  Financial_Account Financial_Account [note: "OR"]
  credit_card credit_card [note: "OR"]
}

Table transaction { 
  id MONGO_ID [pk]
  category Category_Sub
  type string // Expense / Income / Refund / Short Loan (short term to a friend / other) / Deposit / Withdrawal
  amount double
  description string 
}

// END---------- Transactions ----------END \\


// START-------- Loans --------START \\

// Bank loans, Mortgages, Other loans
Table loan { 
  id MONGO_ID [pk]
  goal string
  fromDate datetime
  tillDate datetime
  name string
  lenderName string
  currency Currency
  originalLoanAmount double // Loan amount
  description string 
  loanPath LIST_LoanPath // Can have many paths or 1
}

Table loanPath {
  pathName string
  yearlyInterest double // In %
  totalInterset double // In value (₪ / $..) 
  monthlyPaymentIntrest double 
  monthlyPaymentKeren double 
  monthsLeftToPay int 
}

// END---------- Loans ----------END \\
